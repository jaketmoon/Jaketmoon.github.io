c语言单向链表

发表于2024-01-03|更新于2024-01-03
|阅读量:

跟python一样又完全不一样，一样的是思路一样，不一样的是一个用结构体加结构体指针，一个用数组。

#include<stdio.h>
#include<stdlib.h>
struct student{
    char name[10];
    int score;
    struct student *next;
};
int main(){
    struct student *head=NULL,*pnew,*ptail;
    /**空链表建立头节点**/
    pnew=(struct student*)malloc(sizeof(struct student));/*开辟内存空间*/
    scanf("%s%d",pnew->name,pnew->score);
    head=pnew;/*第一个节点要用头指针指着*/
    ptail=pnew;



    /*在现有的链表添加新节点，此处可以用for循环*/
    pnew=(struct student*)malloc(sizeof(struct student));
    scanf("%s%d",pnew->name,pnew->score);
    ptail->next=pnew;/*连接两个节点*/
    ptail=pnew;/*末节点滚到新的位置*/
    ptail->next=NULL;/*将末节点的下一位赋值*/




    /*遍历链表，可以用for循环*/
    struct student *p;
    p=head;/*从head开始遍历*/
    while(p!=NULL){
        printf("%s%d",p->name,p->score);
        p=p->next;
    }
    



    /*在链表中插入节点，假设链表里的成绩是从高到低排列的，插入后也是如此*/
    p=head;/*便于之后找插入位置辅助节点*/
    struct student *pold;
    pnew=(struct student*)malloc(sizeof(struct student));
    scanf("%s%d",pnew->name,pnew->score);
    if(pnew->score>head->score){
        pnew->next=head;
        head=pnew;
    }else{
        while(p!=NULL&&p->score>pnew->score){
            pold=p;
            p=p->next;
        }/*遍历完成之后p的前一项pold是要插入的节点*/
        /*连接*/
        pnew->next=p;
        pold->next=pnew;
    }

    
    
    
    
    
    /*删除节点,删除Liming的数据,要用free函数解放空间*/
    char xname="Liming";
    p=head;
    if(head->name==xname){
        head=head->next;
        free(p);
        p=head;
    }else{
        while(p!=NULL&&p->name!=xname){
            pold=p;
            p=p->next;
        }
        pold->next=p->next;
        free(p);
    }
    
    return 0;
}
